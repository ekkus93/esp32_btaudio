# web_ui — esp_http_server + embedded gzipped React SPA + REST/WS API (WEB-1).
# The gzipped bundle is embedded by the `main` component (EMBED_FILES); this
# component externs the _binary_index_html_gz_* symbols.
idf_component_register(
    SRCS "web_ui.c"
         "web_ui_wifi.c"
         "web_ui_audio.c"
         "web_ui_radio.c"
         "web_ui_bt.c"
         "web_ui_json.c"
         "web_ui_auth.c"
         "web_ui_auth_core.c"
         "web_ui_ops.c"
    INCLUDE_DIRS "include"
    REQUIRES esp_http_server wifi_mgr bt_link tone radio i2s_out ctrl nvs_flash
    PRIV_REQUIRES json esp_app_format esp_timer esp_timer)

# Embed the gzipped single-file SPA (generated by `web/npm run build` into
# main/www/) directly into THIS library, so web_ui.c's _binary_index_html_gz_*
# externs resolve without cross-component link-order issues. BINARY = no NUL
# terminator (it's gzip bytes, not text).
target_add_binary_data(${COMPONENT_LIB} "${PROJECT_DIR}/main/www/index.html.gz" BINARY)
